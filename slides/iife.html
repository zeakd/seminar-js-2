IIFE Pattern

Imediately Invoked Function Expression


// 익명함수를 생성함과 동시에 실행하여 로컬변수를 분리한다.
// global
(function () {
  // local
})();

// 더글라스 크락포드 방법
(function () {} ());

// 그외 
+function() { ... } ();
-function() { ... } ();
~function() { ... } ();
;function() { ... } ();
!function() { ... } ();

꼭 익혀두자.
